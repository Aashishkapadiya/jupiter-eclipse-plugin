        if (object instanceof IMember) {

          IMember member = (IMember) object;
          this.methodName = member.getElementName();
          ICompilationUnit compilationUnit = member.getCompilationUnit();
          setFieldInstanceValuesFrom(compilationUnit);
          try {
            Document doc = new Document(compilationUnit.getSource());
            ISourceRange range = member.getNameRange();

        }
